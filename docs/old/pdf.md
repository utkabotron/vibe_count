### –®–∞–≥ 1: –°–∫—Ä–∏–ø—Ç (extract\_text.py)

–°–æ–∑–¥–∞–π —Ñ–∞–π–ª `extract_text.py` –∏ –≤—Å—Ç–∞–≤—å —Ç—É–¥–∞ —ç—Ç–æ—Ç –∫–æ–¥. –ù–µ –∑–∞–±—É–¥—å –ø–æ–¥—Å—Ç–∞–≤–∏—Ç—å —Å–≤–æ–π **–Ø–Ω–¥–µ–∫—Å.–¢–æ–∫–µ–Ω**.

```python
import os
import yadisk
import pdfplumber

# --- –ö–û–ù–§–ò–ì–£–†–ê–¶–ò–Ø ---
YANDEX_TOKEN = "–¢–í–û–ô_OAUTH_–¢–û–ö–ï–ù_–û–¢_–Ø–ù–î–ï–ö–°–ê"  # –í—Å—Ç–∞–≤—å —Ç–æ–∫–µ–Ω, –ø–æ–ª—É—á–µ–Ω–Ω—ã–π —Ä–∞–Ω–µ–µ
REMOTE_FOLDER = "/–í—Ö–æ–¥—è—â–∏–µ"
LOCAL_TEMP_DIR = "temp_downloads"

# –°–æ–∑–¥–∞–µ–º –ª–æ–∫–∞–ª—å–Ω—É—é –ø–∞–ø–∫—É –¥–ª—è –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤, –µ—Å–ª–∏ –Ω–µ—Ç
if not os.path.exists(LOCAL_TEMP_DIR):
    os.makedirs(LOCAL_TEMP_DIR)

def connect_to_yandex():
    y = yadisk.YaDisk(token=YANDEX_TOKEN)
    if y.check_token():
        print("‚úÖ –£—Å–ø–µ—à–Ω–æ–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ –Ø–Ω–¥–µ–∫—Å.–î–∏—Å–∫—É")
        return y
    else:
        print("‚ùå –¢–æ–∫–µ–Ω –Ω–µ–≤–∞–ª–∏–¥–µ–Ω")
        exit()

def extract_text_from_pdf(pdf_path):
    """–ß–∏—Ç–∞–µ—Ç PDF –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç —á–∏—Å—Ç—ã–π —Ç–µ–∫—Å—Ç."""
    full_text = ""
    try:
        with pdfplumber.open(pdf_path) as pdf:
            for page in pdf.pages:
                # extract_text() —É–º–Ω—ã–π: –æ–Ω –ø—ã—Ç–∞–µ—Ç—Å—è —Å–æ—Ö—Ä–∞–Ω–∏—Ç—å –æ—Ç—Å—Ç—É–ø—ã –∏ —Ç–∞–±–ª–∏—Ü—ã
                text = page.extract_text()
                if text:
                    full_text += text + "\n"
        return full_text
    except Exception as e:
        print(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ —á—Ç–µ–Ω–∏–∏ PDF: {e}")
        return None

def main():
    client = connect_to_yandex()

    # 1. –ü—Ä–æ–≤–µ—Ä—è–µ–º –ø–∞–ø–∫—É "–í—Ö–æ–¥—è—â–∏–µ"
    print(f"üîç –ü—Ä–æ–≤–µ—Ä—è–µ–º –ø–∞–ø–∫—É '{REMOTE_FOLDER}'...")
    
    # –ü–æ–ª—É—á–∞–µ–º —Å–ø–∏—Å–æ–∫ —Ñ–∞–π–ª–æ–≤ (—Ñ–∏–ª—å—Ç—Ä—É–µ–º —Ç–æ–ª—å–∫–æ —Ñ–∞–π–ª—ã, –Ω–µ –ø–∞–ø–∫–∏)
    files = [f for f in client.listdir(REMOTE_FOLDER) if f.type == 'file' and f.name.endswith('.pdf')]

    if not files:
        print("üìÇ –ü–∞–ø–∫–∞ –ø—É—Å—Ç–∞ –∏–ª–∏ –≤ –Ω–µ–π –Ω–µ—Ç PDF —Ñ–∞–π–ª–æ–≤.")
        return

    # 2. –ë–µ—Ä–µ–º –ø–µ—Ä–≤—ã–π –ø–æ–ø–∞–≤—à–∏–π—Å—è —Ñ–∞–π–ª –¥–ª—è —Ç–µ—Å—Ç–∞
    file_to_process = files[0]
    print(f"üìÑ –ù–∞–π–¥–µ–Ω —Ñ–∞–π–ª: {file_to_process.name}")

    local_path = os.path.join(LOCAL_TEMP_DIR, file_to_process.name)

    # 3. –°–∫–∞—á–∏–≤–∞–µ–º
    print("‚¨áÔ∏è –°–∫–∞—á–∏–≤–∞–µ–º...")
    client.download(file_to_process.path, local_path)

    # 4. –ò–∑–≤–ª–µ–∫–∞–µ–º —Ç–µ–∫—Å—Ç
    print("mining ‚õèÔ∏è –ò–∑–≤–ª–µ–∫–∞–µ–º —Ç–µ–∫—Å—Ç...")
    text_content = extract_text_from_pdf(local_path)

    print("-" * 40)
    print("–†–ï–ó–£–õ–¨–¢–ê–¢ (–ü–µ—Ä–≤—ã–µ 500 —Å–∏–º–≤–æ–ª–æ–≤):")
    print(text_content[:500]) # –í—ã–≤–æ–¥–∏–º –Ω–∞—á–∞–ª–æ, —á—Ç–æ–±—ã –Ω–µ –∑–∞—Å–æ—Ä—è—Ç—å –∫–æ–Ω—Å–æ–ª—å
    print("-" * 40)

    # –í –±—É–¥—É—â–µ–º –∑–¥–µ—Å—å –±—É–¥–µ—Ç –æ—Ç–ø—Ä–∞–≤–∫–∞ text_content –≤ GPT
    
    # –û—á–∏—Å—Ç–∫–∞ (—É–¥–∞–ª—è–µ–º –ª–æ–∫–∞–ª—å–Ω—ã–π —Ñ–∞–π–ª)
    os.remove(local_path)

if __name__ == "__main__":
    main()
```
